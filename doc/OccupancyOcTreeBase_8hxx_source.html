144  template <class NODE>
145  void OccupancyOcTreeBase<NODE>::computeUpdate(const Pointcloud& scan, const octomap::point3d& origin,
146  KeySet& free_cells, KeySet& occupied_cells,
147  double maxrange)
148  {
149 #ifdef _OPENMP
150  omp_set_num_threads(this->keyrays.size());
151  #pragma omp parallel for schedule(guided)
152 #endif
153  for (int i = 0; i < (int)scan.size(); ++i) {
154  const point3d& p = scan[i];
155  unsigned threadIdx = 0;
156 #ifdef _OPENMP
157  threadIdx = omp_get_thread_num();
158 #endif
159  KeyRay* keyray = &(this->keyrays.at(threadIdx));
160  if (!use_bbx_limit) {
161  // -------------- no BBX specified ---------------
162  if ((maxrange < 0.0) || ((p - origin).norm() <= maxrange) ) { // is not maxrange meas.
163  // free cells
164  if (this->computeRayKeys(origin, p, *keyray)){
165 #ifdef _OPENMP
166  #pragma omp critical (free_insert)
167 #endif
168  {
169  free_cells.insert(keyray->begin(), keyray->end());
170  }
171  }
172  // occupied endpoint
173  OcTreeKey key;
174  if (this->coordToKeyChecked(p, key)){
175 #ifdef _OPENMP
176  #pragma omp critical (occupied_insert)
177 #endif
178  {
179  occupied_cells.insert(key);
180  }
181  }
182  } // end if NOT maxrange
183 
184  else { // user set a maxrange and this is reached
185  point3d direction = (p - origin).normalized ();
186  point3d new_end = origin + direction * (float) maxrange;
187  if (this->computeRayKeys(origin, new_end, *keyray)){
188 #ifdef _OPENMP
189  #pragma omp critical (free_insert)
190 #endif
191  {
192  free_cells.insert(keyray->begin(), keyray->end());
193  }
194  }
195  } // end if maxrange
196  }
197 
198  // --- update limited by user specified BBX -----
199  else {
200  // endpoint in bbx and not maxrange?
201  if ( inBBX(p) && ((maxrange < 0.0) || ((p - origin).norm () <= maxrange) ) ) {
202 
203  // occupied endpoint
204  OcTreeKey key;
205  if (this->coordToKeyChecked(p, key)){
206 #ifdef _OPENMP
207  #pragma omp critical (occupied_insert)
208 #endif
209  {
210  occupied_cells.insert(key);
211  }
212  }
213 
214  // update freespace, break as soon as bbx limit is reached
215  if (this->computeRayKeys(origin, p, *keyray)){
216  for(KeyRay::reverse_iterator rit=keyray->rbegin(); rit != keyray->rend(); rit++) {
217  if (inBBX(*rit)) {
218 #ifdef _OPENMP
219  #pragma omp critical (free_insert)
220 #endif
221  {
222  free_cells.insert(*rit);
223  }
224  }
225  else break;
226  }
227  } // end if compute ray
228  } // end if in BBX and not maxrange
229  } // end bbx case
230 
231  } // end for all points, end of parallel OMP loop
232 
233  // prefer occupied cells over free ones (and make sets disjunct)
234  for(KeySet::iterator it = free_cells.begin(), end=free_cells.end(); it!= end; ){
235  if (occupied_cells.find(*it) != occupied_cells.end()){
236  it = free_cells.erase(it);
237  } else {
238  ++it;
239  }
240  }
241  }

245  // early abort (no change will happen).
246  // may cause an overhead in some configuration, but more often helps
247  NODE* leaf = this->search(key);
248  // no change: node already at threshold
249  if (leaf
250  && ((log_odds_update >= 0 && leaf->getLogOdds() >= this->clamping_thres_max)
251  || ( log_odds_update <= 0 && leaf->getLogOdds() <= this->clamping_thres_min)))
252  {
253  return leaf;
254  }
255

257  if (this->root == NULL){
258  this->root = new NODE();
259  this->tree_size++;
260  createdRoot = true;
261  }
262 
263  return updateNodeRecurs(this->root, createdRoot, key, 0, log_odds_update, lazy_eval);

309  template <class NODE>
310  NODE* OccupancyOcTreeBase<NODE>::updateNodeRecurs(NODE* node, bool node_just_created, const OcTreeKey& key,
311  unsigned int depth, const float& log_odds_update, bool lazy_eval) {
312  unsigned int pos = computeChildIdx(key, this->tree_depth -1 - depth);
313  bool created_node = false;
314 
315  assert(node);
316 
317  // follow down to last level
318  if (depth < this->tree_depth) {
319  if (!node->childExists(pos)) {
320  // child does not exist, but maybe it's a pruned node?
321  if ((!node->hasChildren()) && !node_just_created ) {
322  // current node does not have children AND it is not a new node
323  // -> expand pruned node
324  node->expandNode();
325  this->tree_size+=8;
326  this->size_changed = true;
327  }
328  else {
329  // not a pruned node, create requested child
330  node->createChild(pos);
331  this->tree_size++;
332  this->size_changed = true;
333  created_node = true;
334  }
335  }
336 
337  if (lazy_eval)
338  return updateNodeRecurs(node->getChild(pos), created_node, key, depth+1, log_odds_update, lazy_eval);
339  else {
340  NODE* retval = updateNodeRecurs(node->getChild(pos), created_node, key, depth+1, log_odds_update, lazy_eval);
341  // prune node if possible, otherwise set own probability
342  // note: combining both did not lead to a speedup!
343  if (node->pruneNode())
344  this->tree_size -= 8;
345  else
346  node->updateOccupancyChildren();
347 
348  return retval;
349  }
350  }
351 
352  // at last level, update node, end of recursion
353  else {
354  if (use_change_detection) {
355  bool occBefore = this->isNodeOccupied(node);
356  updateNodeLogOdds(node, log_odds_update);
357 
358  if (node_just_created){ // new node
359  changed_keys.insert(std::pair<OcTreeKey,bool>(key, true));
360  } else if (occBefore != this->isNodeOccupied(node)) { // occupancy changed, track it
361  KeyBoolMap::iterator it = changed_keys.find(key);
362  if (it == changed_keys.end())
363  changed_keys.insert(std::pair<OcTreeKey,bool>(key, false));
364  else if (it->second == false)
365  changed_keys.erase(it);
366  }
367  } else {
368  updateNodeLogOdds(node, log_odds_update);
369  }
370  return node;
371  }
372  }

431  template <class NODE>
432  bool OccupancyOcTreeBase<NODE>::getNormals(const point3d& point, std::vector<point3d>& normals,
433  bool unknownStatus) const {
434  normals.clear();
435 
436  OcTreeKey init_key;
437  if ( !OcTreeBaseImpl<NODE,AbstractOccupancyOcTree>::coordToKeyChecked(point, init_key) ) {
438  OCTOMAP_WARNING_STR("Voxel out of bounds");
439  return false;
440  }
441 
442  int vertex_values[8];
443 
444  OcTreeKey current_key;
445  NODE* current_node;
446 
447  int x_index[4] = {-1, 1, 1, -1};
448  int y_index[4] = {1, 1, -1, -1};
449  int z_index[2] = {-1, 1};
450 
451  int k = 0;
452  for(int j = 0; j < 2; ++j){
453  for(int i = 0; i < 4; ++i){
454  current_key[0] = init_key[0] + x_index[i];
455  current_key[1] = init_key[1] + y_index[i];
456  current_key[2] = init_key[2] + z_index[j];
457  current_node = this->search(current_key);
458 
459  if(current_node){
460  vertex_values[k] = this->isNodeOccupied(current_node);
461 
462  // point3d coord = this->keyToCoord(current_key);
463  // OCTOMAP_WARNING_STR("vertex " << k << " at " << coord << "; value " << vertex_values[k]);
464  }else{
465  // Occupancy of unknown cells
466  vertex_values[k] = unknownStatus;
467  }
468  ++k;
469  }
470  }
471 
472  int cube_index = 0;
473  if (vertex_values[0]) cube_index |= 1;
474  if (vertex_values[1]) cube_index |= 2;
475  if (vertex_values[2]) cube_index |= 4;
476  if (vertex_values[3]) cube_index |= 8;
477  if (vertex_values[4]) cube_index |= 16;
478  if (vertex_values[5]) cube_index |= 32;
479  if (vertex_values[6]) cube_index |= 64;
480  if (vertex_values[7]) cube_index |= 128;
481 
482  // OCTOMAP_WARNING_STR("cubde_index: " << cube_index);
483 
484  // All vertices are occupied or free resulting in no normal
485  if (edgeTable[cube_index] == 0)
486  return true;
487 
488  // No interpolation is done yet, we use vertexList in <MCTables.h>.
489  for(int i = 0; triTable[cube_index][i] != -1; i += 3){
490  point3d p1 = vertexList[triTable[cube_index][i ]];
491  point3d p2 = vertexList[triTable[cube_index][i+1]];
492  point3d p3 = vertexList[triTable[cube_index][i+2]];
493  point3d v1 = p2 - p1;
494  point3d v2 = p3 - p1;
495 
496  // Right hand side cross product to retrieve the normal in the good
497  // direction (pointing to the free nodes).
498  normals.push_back(v1.cross(v2).normalize());
499  }
500 
501  return true;
502  }

504  template <class NODE>
505  bool OccupancyOcTreeBase<NODE>::castRay(const point3d& origin, const point3d& directionP, point3d& end,
506  bool ignoreUnknown, double maxRange) const {
507 
509 
510  // Initialization phase -------------------------------------------------------
511  OcTreeKey current_key;
512  if ( !OcTreeBaseImpl<NODE,AbstractOccupancyOcTree>::coordToKeyChecked(origin, current_key) ) {
513  OCTOMAP_WARNING_STR("Coordinates out of bounds during ray casting");
514  return false;
515  }
516 
517  NODE* startingNode = this->search(current_key);
518  if (startingNode){
519  if (this->isNodeOccupied(startingNode)){
520  // Occupied node found at origin
521  // (need to convert from key, since origin does not need to be a voxel center)
522  end = this->keyToCoord(current_key);
523  return true;
524  }
525  } else if(!ignoreUnknown){
526  OCTOMAP_ERROR_STR("Origin node at " << origin << " for raycasting not found, does the node exist?");
527  end = this->keyToCoord(current_key);
528  return false;
529  }
530 
531  point3d direction = directionP.normalized();
532  bool max_range_set = (maxRange > 0.0);
533 
534  int step[3];
535  double tMax[3];
536  double tDelta[3];
537 
538  for(unsigned int i=0; i < 3; ++i) {
539  // compute step direction
540  if (direction(i) > 0.0) step[i] = 1;
541  else if (direction(i) < 0.0) step[i] = -1;
542  else step[i] = 0;
543 
544  // compute tMax, tDelta
545  if (step[i] != 0) {
546  // corner point of voxel (in direction of ray)
547  double voxelBorder = this->keyToCoord(current_key[i]);
548  voxelBorder += double(step[i] * this->resolution * 0.5);
549 
550  tMax[i] = ( voxelBorder - origin(i) ) / direction(i);
551  tDelta[i] = this->resolution / fabs( direction(i) );
552  }
553  else {
554  tMax[i] = std::numeric_limits<double>::max();
555  tDelta[i] = std::numeric_limits<double>::max();
556  }
557  }
558 
559  if (step[0] == 0 && step[1] == 0 && step[2] == 0){
560  OCTOMAP_ERROR("Raycasting in direction (0,0,0) is not possible!");
561  return false;
562  }
563 
564  // for speedup:
565  double maxrange_sq = maxRange *maxRange;
566 
567  // Incremental phase ---------------------------------------------------------
568 
569  bool done = false;
570 
571  while (!done) {
572  unsigned int dim;
573 
574  // find minimum tMax:
575  if (tMax[0] < tMax[1]){
576  if (tMax[0] < tMax[2]) dim = 0;
577  else dim = 2;
578  }
579  else {
580  if (tMax[1] < tMax[2]) dim = 1;
581  else dim = 2;
582  }
583 
584  // check for overflow:
585  if ((step[dim] < 0 && current_key[dim] == 0)
586  || (step[dim] > 0 && current_key[dim] == 2* this->tree_max_val-1))
587  {
588  OCTOMAP_WARNING("Coordinate hit bounds in dim %d, aborting raycast\n", dim);
589  // return border point nevertheless:
590  end = this->keyToCoord(current_key);
591  return false;
592  }
593 
594  // advance in direction "dim"
595  current_key[dim] += step[dim];
596  tMax[dim] += tDelta[dim];
597 
598 
599  // generate world coords from key
600  end = this->keyToCoord(current_key);
601 
602  // check for maxrange:
603  if (max_range_set){
604  double dist_from_origin_sq(0.0);
605  for (unsigned int j = 0; j < 3; j++) {
606  dist_from_origin_sq += ((end(j) - origin(j)) * (end(j) - origin(j)));
607  }
608  if (dist_from_origin_sq > maxrange_sq)
609  return false;
610 
611  }
612 
613  NODE* currentNode = this->search(current_key);
614  if (currentNode){
615  if (this->isNodeOccupied(currentNode)) {
616  done = true;
617  break;
618  }
619  // otherwise: node is free and valid, raycasting continues
620  } else if (!ignoreUnknown){ // no node found, this usually means we are in "unknown" areas
621  OCTOMAP_WARNING_STR("Search failed in OcTree::castRay() => an unknown area was hit in the map: " << end);
622  return false;
623  }
624  } // end while
625 
626  return true;
627  }

733  template <class NODE>
734  std::istream& OccupancyOcTreeBase<NODE>::readBinaryNode(std::istream &s, NODE* node) const {
735 
736  assert(node);
737 
738  char child1to4_char;
739  char child5to8_char;
740  s.read((char*)&child1to4_char, sizeof(char));
741  s.read((char*)&child5to8_char, sizeof(char));
742 
743  std::bitset<8> child1to4 ((unsigned long long) child1to4_char);
744  std::bitset<8> child5to8 ((unsigned long long) child5to8_char);
745 
746  // std::cout << "read: "
747  // << child1to4.to_string<char,std::char_traits<char>,std::allocator<char> >() << " "
748  // << child5to8.to_string<char,std::char_traits<char>,std::allocator<char> >() << std::endl;
749 
750 
751  // inner nodes default to occupied
752  node->setLogOdds(this->clamping_thres_max);
753 
754  for (unsigned int i=0; i<4; i++) {
755  if ((child1to4[i*2] == 1) && (child1to4[i*2+1] == 0)) {
756  // child is free leaf
757  node->createChild(i);
758  node->getChild(i)->setLogOdds(this->clamping_thres_min);
759  }
760  else if ((child1to4[i*2] == 0) && (child1to4[i*2+1] == 1)) {
761  // child is occupied leaf
762  node->createChild(i);
763  node->getChild(i)->setLogOdds(this->clamping_thres_max);
764  }
765  else if ((child1to4[i*2] == 1) && (child1to4[i*2+1] == 1)) {
766  // child has children
767  node->createChild(i);
768  node->getChild(i)->setLogOdds(-200.); // child is unkown, we leave it uninitialized
769  }
770  }
771  for (unsigned int i=0; i<4; i++) {
772  if ((child5to8[i*2] == 1) && (child5to8[i*2+1] == 0)) {
773  // child is free leaf
774  node->createChild(i+4);
775  node->getChild(i+4)->setLogOdds(this->clamping_thres_min);
776  }
777  else if ((child5to8[i*2] == 0) && (child5to8[i*2+1] == 1)) {
778  // child is occupied leaf
779  node->createChild(i+4);
780  node->getChild(i+4)->setLogOdds(this->clamping_thres_max);
781  }
782  else if ((child5to8[i*2] == 1) && (child5to8[i*2+1] == 1)) {
783  // child has children
784  node->createChild(i+4);
785  node->getChild(i+4)->setLogOdds(-200.); // set occupancy when all children have been read
786  }
787  // child is unkown, we leave it uninitialized
788  }
789 
790  // read children's children and set the label
791  for (unsigned int i=0; i<8; i++) {
792  if (node->childExists(i)) {
793  NODE* child = node->getChild(i);
794  if (fabs(child->getLogOdds() + 200.)<1e-3) {
795  readBinaryNode(s, child);
796  child->setLogOdds(child->getMaxChildLogOdds());
797  }
798  } // end if child exists
799  } // end for children
800 
801  return s;
802  }

804  template <class NODE>
805  std::ostream& OccupancyOcTreeBase<NODE>::writeBinaryNode(std::ostream &s, const NODE* node) const{
806 
807  assert(node);
808 
809  // 2 bits for each children, 8 children per node -> 16 bits
810  std::bitset<8> child1to4;
811  std::bitset<8> child5to8;
812 
813  // 10 : child is free node
814  // 01 : child is occupied node
815  // 00 : child is unkown node
816  // 11 : child has children
817 
818 
819  // speedup: only set bits to 1, rest is init with 0 anyway,
820  // can be one logic expression per bit
821 
822  for (unsigned int i=0; i<4; i++) {
823  if (node->childExists(i)) {
824  const NODE* child = node->getChild(i);
825  if (child->hasChildren()) { child1to4[i*2] = 1; child1to4[i*2+1] = 1; }
826  else if (this->isNodeOccupied(child)) { child1to4[i*2] = 0; child1to4[i*2+1] = 1; }
827  else { child1to4[i*2] = 1; child1to4[i*2+1] = 0; }
828  }
829  else {
830  child1to4[i*2] = 0; child1to4[i*2+1] = 0;
831  }
832  }
833 
834  for (unsigned int i=0; i<4; i++) {
835  if (node->childExists(i+4)) {
836  const NODE* child = node->getChild(i+4);
837  if (child->hasChildren()) { child5to8[i*2] = 1; child5to8[i*2+1] = 1; }
838  else if (this->isNodeOccupied(child)) { child5to8[i*2] = 0; child5to8[i*2+1] = 1; }
839  else { child5to8[i*2] = 1; child5to8[i*2+1] = 0; }
840  }
841  else {
842  child5to8[i*2] = 0; child5to8[i*2+1] = 0;
843  }
844  }
845  // std::cout << "wrote: "
846  // << child1to4.to_string<char,std::char_traits<char>,std::allocator<char> >() << " "
847  // << child5to8.to_string<char,std::char_traits<char>,std::allocator<char> >() << std::endl;
848 
849  char child1to4_char = (char) child1to4.to_ulong();
850  char child5to8_char = (char) child5to8.to_ulong();
851 
852  s.write((char*)&child1to4_char, sizeof(char));
853  s.write((char*)&child5to8_char, sizeof(char));
854 
855  // write children's children
856  for (unsigned int i=0; i<8; i++) {
857  if (node->childExists(i)) {
858  const NODE* child = node->getChild(i);
859  if (child->hasChildren()) {
860  writeBinaryNode(s, child);
861  }
862  }
863  }
864 
865  return s;
866  }
